[docs]def summary_lf(self, lf_res=None, times=False, show_opendss_voltages=False):
    """Generates a voltage report for the current power system

    Parameters
    ----------
    lf_res : tuple, optional
        Tuple containing convergence status of the load flow
    times: `bool`, default False
        Print the timing report
    show_opendss_voltages: `bool`, default False
        If True, import and print the voltage profile file generated by OpenDSS.
        A file named <filename>_VLN_Node.Txt should be present in the current
        folder. 
    """

    if lf_res:
        lf_status, lf_it, lf_errv, lf_method = lf_res
        print('\nLoadflow - method', lf_method)

        if lf_status == 0:
            print()
            print('Convergence in', lf_it, 'iterations (err <', lf_errv, ' pu)')
        elif lf_status == 1:
            print()
            print('Convergence not achieved!')  
        print()


    #np.set_printoptions(precision=5, suppress=True, linewidth=110)
    np.set_printoptions(suppress=True, precision=3, linewidth=100)
    grid = self.grid
    filename = self.filename

    if False:
        I0=grid.Y.dot(grid.E)
        print()
        print('Current mismatches:\n')
        print(grid.orderbusn)
        print
        print(abs(I0[3:]-(grid.Il[3:]+grid.Ic[3:])))
        print()             
    
    if show_opendss_voltages:  # load OpenDss voltage references

        vdss = grid.dss.read_vln_vll(os.path.splitext(grid.dss.filepath)[0] + "_VLN_Node.Txt", grid.sub_bus)

        orderbus_dss = list(vdss[vdss['bus'] != '-']['bus']) # order in opendss

        #if grid.sub_bus.upper() in orderbus_dss:
        for b in range(len(grid.orderbus)):
            if grid.orderbus[b].upper() in orderbus_dss:
                orderbus_dss[orderbus_dss.index(grid.orderbus[b].upper())] = grid.orderbus[b]

        orderbusn_dss = [None] * grid.nbus
        cc = 0
        for b, bus in enumerate(orderbus_dss):
            nph = int(grid.busnodes[bus]['nphases'])
            orderbusn_dss[b] = bus + '.' + '.'.join(vdss['node'][cc:cc+nph].astype('str'))
            cc += nph

        E, vbase = grid.reorder_e(orderbusn_dss) # reorder complex voltages based on a given bus order list (*** change to ordern)
        busn_list = np.concatenate([[f'{s[0]}.{i}' for i in s[1:]] for s in (x.split('.') for x in orderbusn_dss)])
        if grid.sbase:
            e_kv = E * vbase /(3**0.5)
            e_pu = E
        else:
            e_kv = E
            e_pu = E / (vbase /(3**0.5))
        diff_v = np.array(1000*vdss['vln-kv'])- abs(e_kv)
        diff_pu = np.array(vdss['vln-pu'])- abs(e_pu)
        diff_ang = np.array(vdss['vln-ang'])-np.angle(E)*180/pi 
        maxabsv_v = abs(diff_v).max()
        maxabsv_pu = abs(diff_pu).max()
        maxabsa = abs(diff_ang).max()        
        report_columns = np.array((np.array(busn_list), vbase/np.sqrt(3) / 1000, abs(e_pu), 
                                   np.array(vdss['vln-pu']), diff_pu, abs(e_kv), np.array(1000*vdss['vln-kv']), 
                            diff_v, np.angle(E)*180/pi, np.array(vdss['vln-ang']), diff_ang))

        report_columns_name = ['bus', 'base_kv', 'vln-pu', 'vln_dss-pu', 'diff-pu', 'vln', 'vln_dss', 'diffv', 'ang', 'ang_dss', 'diffa' ]

    else:

        E, vbase = grid.E, grid.vbase
        busn_list = np.concatenate([[f'{s[0]}.{i}' for i in s[1:]] for s in (x.split('.') for x in grid.orderbusn)])
        if grid.sbase:
            e_kv = E * vbase /(3**0.5)
            e_pu = E
        else:
            e_kv = E
            e_pu = E / (vbase /(3**0.5))
        report_columns = np.array((np.array(busn_list), vbase/np.sqrt(3) / 1000, 
                            abs(e_pu),  abs(e_kv), np.angle(E)*180/pi))
        report_columns_name = ['bus', 'base_kv', 'vln-pu', 'vln', 'ang' ]        

    DE = pd.DataFrame(report_columns, np.array(report_columns_name)).T

    DE["base_kv"] = pd.to_numeric(DE["base_kv"])    
    DE['vln'] = pd.to_numeric(DE['vln'])
    DE['vln-pu'] = pd.to_numeric(DE['vln-pu'])    
    DE["ang"] = pd.to_numeric(DE["ang"])
    DE['vln'] = DE['vln'].map(lambda x: '%9.2f' % x)
    DE['base_kv'] = DE['base_kv'].map(lambda x: '%8.2f' % x)
    DE['vln-pu'] = DE['vln-pu'].map(lambda x: '%7.4g' % x)
    DE['ang'] = DE['ang'].map(lambda x: '%g' % x)    

    if show_opendss_voltages:
        DE["vln_dss"] = pd.to_numeric(DE["vln_dss"])
        DE["vln_dss-pu"] = pd.to_numeric(DE["vln_dss-pu"])
        DE["ang_dss"] = pd.to_numeric(DE["ang_dss"])
        DE["diffa"] = pd.to_numeric(DE["diffa"])
        DE["diffv"] = pd.to_numeric(DE["diffv"])
        DE["diff-pu"] = pd.to_numeric(DE["diff-pu"])
        DE['vln_dss'] = DE['vln_dss'].map(lambda x: '%14.2f' % x)    
        DE['ang_dss'] = DE['ang_dss'].map(lambda x: '%g' % x)
        DE['vln_dss-pu'] = DE['vln_dss-pu'].map(lambda x: '%7.4g' % x)
        DE['diffa'] = DE['diffa'].map(lambda x: '%.2g' % x)
        DE['diffv'] = DE['diffv'].map(lambda x: '%.2g' % x)
        DE['diff-pu'] = DE['diff-pu'].map(lambda x: '%.2g' % x)

        DE.rename(columns={'vln_dss': 'vln ', 'vln_dss-pu': 'vln-pu ', 'ang_dss':'ang '}, inplace=True)
        # Tweaking column header with a fake row
        DE = pd.concat([pd.DataFrame([{'bus':'','base_kv':'','vln-pu':'','vln-pu ':'(OpenDSS)', 'diff-pu':'',
        'vln':'','vln ':'(OpenDSS)','diffv':'','ang':'','ang ':'(OpenDSS)','diffa':''}]), DE], ignore_index=True)
        DE.index = ['',*range(len(DE.index)-1)]        

    #diff_ELL = np.array(( np.array(1000*vv['vll-kv']), abs(grid.Ell), np.array(1000*vv['vll-kv']) - abs(grid.Ell),
    #                      np.array(vv['vll-ang']), np.angle(grid.Ell)*180/pi, np.array(vv['vll-ang']) - np.angle(grid.Ell)*180/pi
    #                      )).T

    pd.set_option('display.max_rows', None)
    pd.set_option("display.precision", 2)
    pd.set_option('display.float_format', lambda x: '%g' % x)
 
    print(DE)
    print()
    if show_opendss_voltages:
        print ('Maximum magnitude error: ', end= '')
        print('{:.5g} p.u. ({:.5g} V)'.format( maxabsv_pu, maxabsv_v) )
        print ('Maximum angle error: {:.5g} deg.'.format(maxabsa) )
        print()

    if times:
        print('Timings report:')
        print()
        grid_times = list(self.grid.times_data.items())
        lf_times = list(self.times_data.items())
        [print('{:s}: {:.4f}s'.format(grid_times[t][0],grid_times[t][1])) for t in [0,1,2]]
        print('{:s}: {:.4f}s'.format(lf_times[0][0],grid_times[0][1]))
        [print('{:s}: {:.4f}s'.format(grid_times[t][0],grid_times[t][1])) for t in [3,4]]
        print()





[docs]def summary_dse (self, status, it, maxabserr, Ikm_true, E_true, 
                 reportmeas=False, reportv=True, reporti=True ):
        """Prints a summary of the state estimation process
        """

        print('\nBranch-current based Distributed State estimation - report\n')
        print('System', self.grid.sysname)
        print()
        print('Number of buses:', self.grid.nbus)
        print('Number of nodes:', self.grid.nnodes)
        print('Number of branches:', self.grid.nbr)
        print()
        print('Number of measurements:', self.meter.nm())
        print('Number of ACTIVE measurements:', self.meter.nma())
        print('Number of states:', self.N)

        if reportmeas:
            print(self.meter)

        iteration = 'iteration'
        if it > 1:
            iteration += 's'    

        if self.isObservable:
            if status == 0:
                print()
                print('Convergence in', it, iteration, '(err <', maxabserr, ' A)')
                #print('Maxabserr =', maxabserr)
                #print('abs(Ikm_true-self.Ikm)')
                #print(abs(Ikm_true-self.Ikm))

            elif status == 1:
                print()
                print('Convergence not achieved after', it, iteration)  
        else:
                print('The system', self.grid.sysname , 'is not observable!\n')
                print("The following branches aren't observable:")
                print()
                print(list(np.array(self.states)[self.nonobstates]))
                print()
                print('************\n')
                return
        print()

        # Printing branch currents
        if len(self.grossmeas) > 0:
            print('*******')
            print("The following measurements have been deactivated "
                  " due to gross errors: ")
            print()
            print('{:10s} {:8s}'.format('Meas.', 'Normalized residual (r_)'))
            for m in range(len(self.grossmeas)):
                zstr = np.array(self.Zstr)[self.grossmeas[m]]
                r_ = self.grossmeas_r_[m]
                print('{:10s} {:.8g}'.format(zstr, r_))
            print()
            print('*******')
            print()


        if reporti:
            br_list = []
            for k in self.indexbr.keys():
                for m in self.indexbr[k].keys():
                    for e, n in enumerate(self.indexbr[k][m]['nodes']):
                        if e ==0:
                            br_list += [k + '-' + m + ' ' + n ]
                        else:
                            br_list += [len(k + '-' + m + ' ')*' ' + n ]
            
            print('{:^10} {:^12} {:^12} {:^12}'.format('branch','|Ikm_true|','|Ikm_est|', 'err (A)'))
            print(''.join(['-']*47))
            for i in range(len(br_list)):
                #print('{} {:18.4g} {:18.4g} {:10.4g}'.format(br_list[i], Ikm_true[i], self.Ikm[i], abs(Ikm_true[i]-self.Ikm[i])))
                print('{:10} {:^12.4g} {:^12.4g} {:^12.4g}'.format(br_list[i], 
                    abs(Ikm_true[i]), 
                    abs(self.Ikm[i]), 
                    abs(Ikm_true[i]-self.Ikm[i])))
            print()
        
        maxerri = abs(Ikm_true-self.Ikm).max()
        print('Maximum current error: {:.4g} A'.format(maxerri))
        print()        

        # Printing voltages at nodes
        if reportv:
            print()
            print('Voltage profile:')
            print()
            ii = 0
            print('{:^6}  {:^11} {:^11} {:^11}'.format('bus','kV_true','kV_est', 'err (V)'), end='')
            print('   {:^11} {:^11} {:^11}'.format('Vpu_true','Vpu_est', 'err_pu'))
            print(''.join(['-']*83))
            for k in self.grid.busnodes.keys():
                for e, n in enumerate(self.grid.busnodes[k]['nodes']):
                    if e == 0:
                        print('{:4s}{:>2s}  '.format(k,n),end='')
                    else:
                        print('{:>6s}  '.format(n),end='')
                    
                    print('{:^11.6g} {:^11.6g} {:^11.6g}    {:^11.4g} {:^11.4g} {:^11.4g}'.format(
                        abs(E_true[ii]) , 
                        abs(self.grid.E[ii]), 
                        abs(E_true[ii]-self.grid.E[ii]),
                        abs(E_true[ii]) / (self.grid.vbase[ii]/3**0.5), 
                        abs(self.grid.E[ii]) / (self.grid.vbase[ii]/3**0.5), 
                        abs(E_true[ii]-self.grid.E[ii])/ (self.grid.vbase[ii]/3**0.5)))                   
                    ii += 1
            print()

        maxerrpu = (abs(E_true-self.grid.E)/(self.grid.vbase/3**0.5)).max()        
        maxerrv = (abs(E_true-self.grid.E)).max()
        print('Maximum voltage error: {:.4g} p.u. ({:.4g} V)'.format(maxerrpu, maxerrv))

        #print()
        #rri = abs(np.concatenate((self.r_.real,self.r_.imag)))
        #print('Maximum normalized residual: ', rri.max())
        #print('Maximum absolute: ', abs(self.r_).max())
