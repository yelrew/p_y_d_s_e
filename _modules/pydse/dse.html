    """A class to Estimate the State of unbalanced Distribution Systems

    Currently, the Branch-current method is implemented, using a Sparse
    solver from `scipy`. Artificial measurements can be generated from
    the `zbus` load-flow method.

    The observability check is performed by analyzing the transpose of
    the Jacobian Matrix. If the system is not observable, a list of the
    relevant branches that need additional measurements is provided.


    Attributes
    ----------
    grid : `powergrid`
       Power system represented in a Powergrid object.
       If not provided, a grid object will be generated based on `filename`
    filename : `string`, optional
        Optional filename to instantiate the Powergrid object
    loadflow : `Loadflow` 
        If measurement generation is requested, holds the loadflow object
    meter : list [`Meter`]
        Stores the measurements available in a `Meter` object
    indexbr: `dict` [`str`, `dict`]
        Associative array with index information for each branch km.
        The key represents bus `k`, and the values hold bus `m` and the location
        of the branch in the state vector `Ikm`.  See :py:func:`~pydse.dse.Dse.indexBranches` 
        for more details.
    method: `str`, default 'branch-current'
        Solution method of the State Estimator.

    nmt : `int`
        Number of meters
    nbr : `int`
        Number of branches
    N : `int`
        Number of state variables
    M : `int`
        Number of measurements

    Z : `numpy.array`, (M, )
        Measurement (complex) vector
    H : 2D `numpy.array`, (M, N)
        Dense Jacobian Matrix        
    Hs : `sparse.csc_matrix`, (M, N)
        Sparse Jacobian Matrix (column-major)
    Ws : `sparse.coo_matrix`, (M, M)
        Sparse Weighting Matrix        
    G : `sparse.csc_matrix`, (M, N)
        Gain matrix
    Gf : `sparse.csc_matrix`, (M, N)
        Factorized gain matrix
    r : `numpy.array`, (M, )
        Residual (complex) vector
    r_ : `numpy.array`, (M, )
        Normalized residual (complex) vector        
    r_thres : `float`, default 4.
        Normalized residual test threshold    
    grossmeas : `list` [`int`]
        Index of the measurements flagged with gross errors.
        See :py:func:`~pydse.dse.Dse.checkGrossErrors`
    grossmeas_r_ : `list` [`float`]
        Normalized residuals of the erroneous measurement.
    Ikm : `numpy.array` , (M, )
        Complex branch-current state vector (estimated currents)
    E : `numpy.array` 
        Complex nodal voltages vector.
    states : `list` [`str`]
        String-list representation of the states   
    Zstr : `list` [`str`]
        String-list representing the measurement vector
    zmap : `list` [`np.ndarray`]
        Stores the location (position in the meter list and the internal node) 
        of the measurement Z inside the meter structure.
    isObservable : `bool` 
        Observability status of the system. True for an obsevable system.
        See :py:func:`~pydse.dse.Dse.checkObservability`
    hasGrossErrors : `bool` 
        True if there are gross errors present.
        Initially, set to True. See :py:func:`~pydse.dse.Dse.checkGrossErrors`
    nonobstates : `list` [`int`]
        Index of the non-observable branches
    tol : `float`, default 1e-8
        Tolerance for the estimator convergence.
    maxit : `int`, default 100
        Maximum number of iterations.
    """

        self.N = None # number of state variables
        self.M = None # number of total measurements
        self.Z = None # Measurement vector
        self.r = None # Residual vector
        self.r_ = None # Normalized residual vector

        self.Ws = None # Sparse Weighting Matrix
        self.E = None # nodal voltages
        self.states = None # string representation of the states
        self.Zstr = None # string list representation of the measurement vector
        self.zmap = [] # mapping between Z and the corresponding meter
        self.nonobstates = None # Will hold the non-observable states
        self.tol = tol # Convergence tolerance
        self.maxit = maxit # Maximum number of iterations
        self.isObservable = False
        self.hasGrossErrors = True
        self.grossmeas_r_ = [] # normalized residuals of the erroneous measurements

[docs]    def addMeas_Realtime(self, type, location, nodes, value, accuracy=None, active=True):
        """Add a real-time measurement

        Currently, the following measurements are allowed: 'skm' (complex power flow) 
        and 's' (complex pseudo-load). The measurement values should be provided 
        in real unities (e.g. kW).

        Parameters
        ----------
        type : string, {'ikm','skm','i','s','|ikm|'}, default 'ikm'
            Measurement type. 
        location : `str`
            The identifier of the bus or branch associated with the measurement            
        nodes : `str` 
            Nodes to which the measurement are connected to. Example, '1', or '123'.
            More than one node indicates a three-phase measurement.            
        value : `float`
            Measurement value (actual value in kW or MW). Three-phase values should
            be entered in a list.
        accuracy : float, optional
            Measurement accuracy (or meter reliability), by default None.

        Note
        ----
        :py:func:`~pydse.dse.Dse.addMeas_Realtime` should be deprecated in the future. 
        Use :py:func:`~pydse.dse.Dse.addMeas`  instead.        

        Examples
        --------
        Adds a real-time (three-phase) power flow measured in branch 632-670.        

        >>> from pydse.dse import Dse
        >>> dse = Dse(filename="data/IEEE13mod.dss")
        >>> dse.addMeas_Realtime('skm', location='632-670', nodes='123',
        value=[1.06725092e6+0.48933956e6j, 524.04365e3+125.60266e3j, 1.10045821e6+0.36163657e6j])

        Adds the complex power drawn by the load at bus 611 and node 3.
        
        >>> from pydse.dse import Dse
        >>> dse = Dse(filename="data/IEEE13mod.dss")
        >>> dse.addMeas_Realtime('s', location='611', nodes='3', value=163.5e3-15.54e3j)
        """        
        # Checking consistency of the measurement location
        
        self.grid.busbranch()
        type = type.lower()
        if type == 's' or type == 'i':
            if location not in self.grid.busnodes:
                print('Bus', location, 'not found in the system')
                exit(0)            
            if not isinstance(location, list):
                location = location.split()
        elif type == 'skm' or type == 'ikm' or type == '|ikm|':
            if '-' in location:
                k, m = location.split('-')
                if not self.grid.H.has_edge(k, m):
                    print('Branch between buses', k, 'and', m, 'not found in the system.')
                    exit(0)
            else:
                if location not in self.grid.branches:
                    print('Branch', location, 'not found in the system.')
                    exit(0)
                k = self.grid.branches[location]['bus1'].split('.')[0]
                m = self.grid.branches[location]['bus2'].split('.')[0]
            location = [k,m]
        else:
            print('addMeasRT: can\'t process measurement of type', type)
            exit(1)

        if not isinstance(value, list):
            value = [value]

        # Adding measurement       
        self.meter.add(type=type, value=value, location=location, nodes=nodes, 
                       accuracy=accuracy, active=active) 




[docs]    def addMeas(self, type='ikm', location='all', nodes='all', meas_filepath=None,
                value=None, accuracy=None, active=True, ignore=None, grosserror=0.):
        """Add measurements (real or artificial) for State Estimation

        The measurements can be of power or current types (flow or injection), with 
        values manually inserted or automatically read from a text file, if 
        ``meas_filepath`` is provided. It neither the value nor the data file is provided, 
        an artificial value will be generated from the Load Flow.

        Parameters
        ----------
        type : string, {'ikm','skm','i','s','|ikm|'}, default 'ikm'
            Type of measurement to add.
            Injection measurements with negative signal. 
        nodes : string, {'123', '12'}, default 'all'
            Nodes to which the measurement are connected to. It defaults to 'all', meaning
            all available nodes.
        location : {'bus', 'bus1-bus2', 'branch'}, default 'all'
            Location to place the measurement. If bus, the bus number or a range of buses.
            should be provided. If branch measurement, the terminal buses should be provided, 
            separated by a dash. The full branch id is also accepted. 
            When adding artificial measurements, multiple locations are supported by separating 
            the locations with a comma.
        value : `list` [`float`]
            Measurement values (actual values in kW or MW). Three-phase values should
            be entered in a list. If None, the values will be calculated from load flow.
        meas_filepath : string, default None
            A path to a text file in `sdf` format, containing measurement data.
        accuracy : float, default None.
            Accuracy class of the measurement, for State Estimation and Bad Data detection purposes,
            in per-unit. Typical values are `0.01` or `0.1`, for meter classes 1% and 10%, 
            respectively. See the ``Notes``.
        active : bool, default True
            If true, the measurement is active            

        **Options for the generation of artificial measurements**

        ignore : {'src', 'bus'}, optional
            Buses to be ignored when generating the measurements, by default None.    
        grosserror : `float`, optional, by default 0.
            Gross error (in terms of standard deviations) to be added to the measurement. 
            See the ``Notes``.

        Note
        ----
        Measurements with values are added through the function 
        :py:func:`~pydse.dse.Dse.addMeas_Realtime` (to be deprecated).

        Notes
        -----
        1. Artificial measurements can be created for specific groups of buses/branches or nodes.
        2. By default, a new measurement is always considered 'active' for State Estimation purposes.
        3. The ``accuracy`` field reflects the reliability degree of the associated meter. For State
           Estimation and Bad Data detection purposes, the measurement's standard deviation is 
           assumed to be 1/3 of the ``accuracy``.
           For artificial measurements, the ``accuracy`` is by default 1/3 of the measurement accuracy.

        Examples
        --------
        Add the power flows measured in the line between buses 632 and 670, for all nodes.        

        >>> dse.addMeas('skm', location='632-670', nodes='123',
                value=[1.06725092e6+0.48933956e6j, 524.04365e3+125.60266e3j, 
                       1.10045821e6+0.36163657e6j])

        Same as before, except that the actual branch name is provided.

        >>> dse.addMeas('skm', location='650632', nodes='123',
                value=[1.252e+06+6.85e+05j, 9.728e+05+3.792e+05j, 1.342e+06+6.72e+05j])

        Add a power measurement of the load in bus 611, node 3.

        >>> dse.addMeas('s', location='611', nodes='3', value=163.5e3-15.54e3j)

        Add artificial power measurements from load flow to all buses except bus 'src', at 
        all nodes available in each bus.
    
        >>> dse.addMeas('s', location='all', nodes='src', ignore='src')

        Add power measurements from LF to buses 611, 684, 652, 671, 680, 692, 675, 670:

        >>> dse.addMeas('s', location='611, 684, 652, 671, 680, 692, 675, 670')

        Add power flow measurements from LF to the transformers XFM1, Sub, and branch 650-632:

        >>> dse.addMeas('Skm', location='XFM1, 650-632, Sub')

        Add a flow measurement from LF to branch 645-646, at nodes 2 and 3, with 10% accuracy:
        
        >>> dse.addMeas('skm', location='645-646', nodes='23', accuracy=0.1)

        Add a power measurement from LF to bus 611, with gross error 0.9 
        (90% times the expected accuracy)

        >>> dse.addMeas('s', location='611', accuracy=0.2, grosserror= 0.9 )
        """
        if meas_filepath:
            # If `meas_filepath` is given, read the SDF measurement file
            self.meas_filepath = meas_filepath
            self._addMeasFromSDFfile(meas_filepath)
            return

        if value != None:
            # If the measurement value is given, redirect to `addMeas_Realtime`
            self.addMeas_Realtime(type=type, location=location, nodes=nodes, 
                                  value=value, accuracy=accuracy, active=active)
            return

        # If neither a meas_filepath nor a value is given, pass through (generate a meas. from LF)

        if self.loadflow == None:
            self.loadflow = Loadflow(self.grid) #, loadmodel='1')
            self.loadflow.loadflow(method='zbus')
            #self.grid.calc_Il()
        
        type = type.lower()
        grid = self.grid
        std_noise = 0. if not accuracy else accuracy/3.
        
        # LF - Bus measurements

        if type == 's'  or type == 'i':

            location = location.replace(' ','').split(',') # handles - and ,
            buslist = [] # list of buses from which the measurements will be created
            for loc in location:
                if loc == 'all':
                    buslist += list(self.grid.busnodes.keys())
                elif loc == 'load':
                    buslist += sorted(list(set([grid.load[load]['bus1'].split('.')[0] for load in grid.load])))
                else: # TODO: add bus range handling
                    buslist += [loc]            
            if ignore:
                ignore = ignore.replace(' ','').split(',')
                ignorelist = []
                for igbus in ignore:
                    if igbus == 'src' and grid.sub_bus in buslist:
                        ignorelist += [grid.sub_bus]
                    elif igbus == 'prim': # ignore transformer's primary buses
                        ignorelist += grid.trbus1
                    elif igbus == 'sec': # ignore transformer's primary buses
                        ignorelist += grid.trbus2
                    else:
                        ignorelist += [igbus]            
                buslist = list(set(buslist)-set(ignorelist))

            # Power measurement (load)
            if type == 's':
                grid_meas = -self.grid.E * np.conj(grid.Il + grid.Ic) # S will be kept fixed

            # Current injection measurement (net bus flow)
            elif type == 'i':

                grid_meas = np.zeros(self.grid.nnodes, dtype=complex)

                for branch in self.grid.branches:
                    k = self.grid.branches[branch]['bus1']
                    m = self.grid.branches[branch]['bus2']
                    knodes = self.grid.branches[branch]['nodes1']
                    mnodes = self.grid.branches[branch]['nodes2']
                    iknodes = [self.grid.busnodes[k]['nodes'].index(n) for n in knodes]
                    imnodes = [self.grid.busnodes[m]['nodes'].index(n) for n in mnodes]
                    ik = np.array(self.grid.busnodes[k]['inodes'])[iknodes]
                    im = np.array(self.grid.busnodes[m]['inodes'])[imnodes]
                    ikm = self.grid.branches[branch]['ikm']
                    grid_meas[im] -= ikm
                    grid_meas[ik] += ikm

            else:
                print('addMeas: Can''t recognize measurement of type', type)
                
            for bus in buslist:                 
                si = self.grid.busnodes[bus]['si'] # starting index
                if nodes == 'all':
                    nodes_ = self.grid.busnodes[bus]['nodes']
                    nph = self.grid.busnodes[bus]['nphases']                    
                    value = grid_meas[si:si+nph] * (1 + np.random.normal()*std_noise + grosserror)
                else:
                    nodes_ = nodes
                    value = np.zeros(len(nodes), dtype=complex)
                    for e, n in enumerate(nodes):
                        k = self.grid.busnodes[bus][n]
                        value[e] = grid_meas[k]* (1 + np.random.normal()*std_noise + grosserror)

                self.meter.add(type=type, value=value, nodes=nodes_, location=[bus], 
                                    accuracy=accuracy, active=active ) 

        # LF - Branch measurements
        
        elif type == 'skm' or type == 'ikm' or type == '|ikm|':            
            
            location = location.replace(' ','').split(',') # remove blank spaces and split strings
            branch_list = []
            for loc in location:
                if loc == 'all':
                    branch_list += self.grid.branches
                elif loc == 'sec': # currents flowing out of the TRs secondary sides
                    for id_ in self.grid.branches:
                        if self.grid.branches[id_]['bus2'] in self.grid.trbus2:
                            branch_list += [id_]
                elif loc == 'prim': # currents flowing into the TRs primary side
                    for id_ in self.grid.branches:
                        if self.grid.branches[id_]['bus1'] in self.grid.trbus1:
                            branch_list += [id_]
                else:
                    if '-' in loc:
                        k, m = loc.split('-')
                        branch_list += [self.grid.H[k][m][0]['id_']]
                    else:
                        branch_list += [loc]

                if ignore:
                    ignore_list = ignore.replace(' ','').split(',')
                    ignore = []
                    for ig in ignore_list:
                        if '-' in ig: # fetching branch id_ from terminal buses
                            k, m = ig.split('-')
                            ignore += [self.grid.H[k][m][0]['id_']]
                        else:
                            ignore += [ig]

                    branch_list = list(set(branch_list)-set(ignore))

            for branch in branch_list:
                k = self.grid.branches[branch]['bus1'].split('.')[0]
                m = self.grid.branches[branch]['bus2'].split('.')[0]
                grid_meas = self.grid.branches[branch][type]

                if nodes == 'all':
                    nodes_ = self.grid.branches[branch]['nodes2']
                    if nodes_ != self.grid.branches[branch]['nodes1']:
                        print('addMeas: nodes1 and nodes2 are different (measurement', type, ')')
                        exit(1)
                    value = grid_meas * (1 + np.random.normal()*std_noise + grosserror)
                else:
                    nodes_ = nodes
                    dtype = None if type == '|ikm|' else complex
                    value = np.zeros(len(nodes), dtype=dtype)
                    for v, n in enumerate(nodes):
                        if n not in self.grid.branches[branch]['nodes2']:
                            nodes_ = nodes_.replace(nodes[v],'')
                            continue
                        j = self.grid.branches[branch]['nodes2'].index(n)
                        value[v] = grid_meas[j] * (1 + np.random.normal()*std_noise + grosserror)
                if len(nodes_) == 0:
                    continue

                self.meter.add(type=type, value=value, nodes=nodes_, location=[k, m], 
                                    accuracy=accuracy, active=active)


    def _addMeasFromSDFfile(self, meas_filepath):
        """Read measurements from a SMF file

        Parameters
        ----------
        meas_filepath : str
            A path to a text file in `sdf` format, containing measurement data.
        
        Notes
        -----
        The `sdf` format is a text file with the following measurement fields
        separed by lines: key, bus1, bus2, nodes kw, kvar, amp        
        """       
        KEY_DSE_MEAS_TYPE_SKM: int = 1
        KEY_DSE_MEAS_TYPE_S_PSEUDO: int = 2 
        KEY_DSE_MEAS_TYPE_IKM: int = 3

        meas_data = np.loadtxt(meas_filepath, 'str').reshape(-1,7)

        for row in meas_data:
            key, bus1, bus2, nodes, kw, kvar, amp = row            
            type = {KEY_DSE_MEAS_TYPE_SKM: 'skm', 
                    KEY_DSE_MEAS_TYPE_S_PSEUDO: 's', 
                    KEY_DSE_MEAS_TYPE_IKM: '|ikm|'
                }.get(int(key), None)
            location = bus1 + bus2 if type != 's' else bus1
            
            # Implementing algebric operators in the value field
            value = dict.fromkeys(['kw', 'kvar', 'amp'])
            for field in ['kw', 'kvar', 'amp']:
                data = eval(field)
                if '*' in data:
                    mult = data.strip().split('*')
                    data = float(mult[0])*float(mult[1])
                value[field] = data                            

            value = float(value['amp']) if type == '|ikm|' \
                    else (float(value['kw']) + float(value['kvar'])*1j)*1000            
            # accuracy = 0.01 if type == '|ikm|' else None

            if not type:
                print('Warning \'_addMeasFromSDFfile\': ignoring measurement of type', type)
                continue

            self.addMeas_Realtime(type, location=location, nodes=nodes, value=value)

[docs]    def indexBranches(self):
        """ Construct the index containing the branch order for BcDSE

        The index is stored in the 3-level dictitionary ``indexbr``, with the first two 
        keys holding the terminal buses of the branch, and the last key storing the 
        absolute location of each individual node in the  the state vector `Ikm`.
        """
        b = 0 # branch counter
        self.states = []
        for k, m in self.grid.sweepfromroot.edges:
            if k not in self.indexbr:
                self.indexbr[k] = {}
            self.indexbr[k][m] = {}
            # Union of the phases of all wires connected between the buses k and m
            branch_data = self.grid.H.get_edge_data(k,m).items()
            nodes = ''.join(([item[1]['nodes2'] for item in branch_data]))
            id_ = [item[1]['id_'] for item in branch_data]
            #for node in self.grid.branches[id_]['nodes2']:
            self.indexbr[k][m]['nodes'] = nodes
            self.indexbr[k][m]['indexes'] = [*range(b,b+len(nodes))]
            self.indexbr[k][m]['ids'] = id_
            for node in nodes:
                self.indexbr[k][m][node] = b
                self.states += [ str(k) + '-' + str(m) + '@' + node ]
                b += 1
                #self.indexbr[k][m]['zkm'] = zkm  # TODO: much safe to extract zkm from Y

        self.nbr = b
        self.N = b # defining here the number of state variables




[docs]    def getTrueCurrents(self):
        """Return the true branch current vector from the load flow results

        Returns
        -------
        Ikm_true: ndarray
            Array of the true branch currents evaluted from load flow
        """
        Ikm_true = np.zeros(self.grid.nbrp, dtype=complex)        

        for id_ in self.grid.branches: # iterating over the index
            k = self.grid.branches[id_]['bus1']
            m = self.grid.branches[id_]['bus2']
            knodes = self.grid.branches[id_]['nodes1']
            mnodes = self.grid.branches[id_]['nodes2']
            ikm = self.grid.branches[id_]['ikm']
            if knodes != mnodes:
                print('getTrueCurrents: knodes and mnodes in different order')
                exit(1)
            for v, n in enumerate(mnodes):
                Ikm_true[self.indexbr[k][m][n]] = ikm[v]
        
        return Ikm_true




[docs]    def formZ(self):
        """Form the measurement vector (Z)

        The vector Z contains only measurements of current type.
        Power measurements are converted to pseudo-current measurements
        using the voltages available.

        Note
        ----
        Vector Z is build considered only ACTIVE measurements.
        If a measurement is flagged as gross error after the estimation
        process, it will be deactivated.
        See :py:func:`~pydse.dse.Dse.checkGrossErrors` for more details.
        """ 
        self.nmt = self.meter.nmt()        
        zmap = [] 
        E = self.grid.E
        Zstr = [] 
        Z = []

        for imeter, meas in enumerate(self.meter.meter):
            bus = meas.location
            nodes = meas.nodes
            value = meas.value.copy()
            type_ = meas.type
            loc = meas.location
            active = meas.active

            for v, node in enumerate(nodes): # spliting measurements by nodes, here                
                
                if not active[v]:
                    continue
                k = self.grid.busnodes[bus[0]][node]     

                # Converting pseudo- complex power to current, if needed
                if type_ == 's': 
                        value[v] = np.conj(value[v] / E[k] )
                elif type_ == 'skm':
                    if bus[1] in self.grid.trbus2: 
                        p = self.grid.busnodes[bus[1]][node] # tr virtual bus
                        value[v] = np.conj(value[v] / (E[p])) # ikm = conj(skm / Ep)
                    else:
                        value[v] = np.conj(value[v] / (E[k])) # ikm = conj(skm / Ek)
                elif type_ == 'v':
                    print('setZi: can\'t handle voltages measurements yet')
                    exit(1)
                elif type_ == '|ikm|':
                    if self.has_nonlinear_meas == False:
                        self.has_nonlinear_meas = True
                
                if type_ in ['|ikm|']: # future include |v| 
                    Z += [value[v]]
                    zmap += [(imeter, node)]
                    Zstr += ['|' + type_[1] + '-'.join(list(loc)) + '@' + node + '|']

                if type_ in ['ikm', 'skm', 'i', 's']:
                     Z += [value[v].real, value[v].imag]
                     zmap += [(imeter, node), (imeter, node)]
                     Zstr += [type_[0] + '-'.join(list(loc)) + '_r@' + node,
                              type_[0] + '-'.join(list(loc)) + '_i@' + node]

        self.Zstr = Zstr
        self.Z = np.array(Z)
        self.zmap = zmap




[docs]    def formH(self):
        """Form the Jacobian Matrix (H) for State Estimation
        """    
        N = self.N
        hrow, hcol, hdata = [], [], [] # row, column and value of sparse matrix

        cm = 0 # measurement counter
        for meas in self.meter.meter:
            type = meas.type
            nodes = meas.nodes
            active = meas.active

            if type == 'ikm' or type == 'skm':
                k, m = meas.location
                for n_, node in enumerate(nodes):
                    if not active[n_]:
                        continue
                    n = self.indexbr[k][m][node] # index in the state vector
                    hrow += [cm]; hcol += [n]; hdata += [1] # ∂Ikm/Ikm_r                     
                    hrow += [cm + 1]; hcol += [n + N]; hdata += [1] # ∂Ikm/Ikm_i
                    cm += 2

            elif type == '|ikm|' :
                k, m = meas.location
                for n_, node in enumerate(nodes):
                    if not active[n_]:
                        continue
                    n = self.indexbr[k][m][node] # index in the state vector
                    Ikm_phase = np.angle(self.Ikm[n])
                    hrow += [cm]; hcol += [n]; hdata += [np.cos(Ikm_phase)]
                    hrow += [cm]; hcol += [n + N]; hdata += [np.sin(Ikm_phase)]
                    cm += 1               

            elif type == 's' or type == 'i':
                k = meas.location[0]
                predecessors = list(self.grid.sweepfromroot.predecessors(k)) # back to the substation
                successors = list(self.grid.sweepfromroot.successors(k)) # after th

                for n_, node in enumerate(nodes):
                    if not active[n_]:
                        continue
                    for m in successors:
                        if node not in self.indexbr[k][m]:
                            continue # node doesn't exist in current branch
                        n = self.indexbr[k][m][node]
                        if type == 's':
                            hdata += [-1, -1]
                        else:
                            hdata += [1, 1]
                        hrow += [cm]; hcol += [n] 
                        hrow += [cm + 1]; hcol += [n + N]

                    for i in predecessors:
                        if node not in self.indexbr[i][k]:
                            continue # node doesn't exist in current branch
                        n = self.indexbr[i][k][node]
                        if type == 's':
                            hdata += [1, 1]
                        else:
                            hdata += [-1, -1]
                        hrow += [cm]; hcol += [n]
                        hrow += [cm + 1]; hcol += [n + N]
                        
                    cm += 2
        
        self.Hs = sparse.coo_matrix((hdata, (hrow, hcol))).tocsc()
        self.M = self.Hs.shape[0]




[docs]    def formW(self):
        """Construct the Weighting Matrix (W)

        The weight of each measurement is computed as the reciprocal of the
        correponding variance.
        """

        Wd, Wdata = [], [] # diagonal and data

        for meas in self.meter.meter:
            if meas.accuracy == None:
                std = 1.
            else:
                std = meas.accuracy / 3

            for v in range(len(meas.nodes)): # spliting measurements by nodes, here
                if not meas.active[v]:
                    continue
                Wd += [len(Wd)]
                Wdata += [1/(std**2)]
                if meas.type in ['skm', 'ikm', 's', 'i']:
                    # Apply the same weights for real and imaginary measurements.
                    Wd += [len(Wd)]
                    Wdata += [1/(std**2)]

        self.Ws = sparse.coo_matrix((Wdata, (Wd, Wd)))        




[docs]    def updateVoltages(self):
        """Update the nodal Voltages using the available branch-currents

        This is the step 2 of the BrcDSE algorithm. It performes a forward
        sweep starting from the substation, updating all nodal voltages
        using Kirchhoff's Voltage Law.

        """
        E = self.grid.E

        for k, m in self.grid.sweepfromroot.edges:
            indexes = self.indexbr[k][m]['indexes']
            ids = self.indexbr[k][m]['ids']
            
            for id_ in ids: # handling parallel branches
                #type = self.grid.branches[id_]['type']
                nodes = self.grid.branches[id_]['nodes2']
                indexes = [self.indexbr[k][m][n] for n in nodes]
                iknodes = [self.grid.busnodes[k]['nodes'].index(n) for n in nodes]
                ik = np.array(self.grid.busnodes[k]['inodes'])[iknodes]
                imnodes = [self.grid.busnodes[m]['nodes'].index(n) for n in nodes]
                im = np.array(self.grid.busnodes[m]['inodes'])[imnodes]
                zkm = self.grid.branches[id_]['zkm']
                akm = self.grid.branches[id_]['akm']
                At = self.grid.branches[id_]['At']
                ikm = self.Ikm[indexes]
                # Updating terminal voltage
                E[im] = At.dot(E[ik]) - zkm.dot(ikm)*akm**2 # KVL




[docs]    def checkObservability(self):
        """Checks the network observability by analyzing matrix Ht

        The method is based on the factorization of the transpose of the 
        Jacobian Matrix (H.T). If any row of the factorized matrix has all
        elements zero, it indicates the corresponding state is non-observable.
        """
       
        # ht = self.H.T
        ht = self.Hs.todense().T

        N, M = ht.shape

        if M < N:
            # ht = np.hstack((ht,np.tile(ht[:,-1],(N-M,1)).T))
            ht = np.hstack((ht,np.tile(ht[:,-1].reshape(-1,1),(N-M))))

        for j in range(N):
            if abs(ht[j,j]) < 1e-20:
                continue
            for i in range(j+1, N):                
                if abs(ht[i,j]) > 1e-20:
                    cij = -ht[i,j]/ht[j,j]
                    for k in range(N):
                        ht[i,k] += ht[j,k]*cij

        sumrows = np.zeros(N)
        for n in range(N):
            for m in range(M):
                sumrows[n] += abs(ht[n,m])
        
        self.nonobstates = np.where(sumrows==0)[0]
        
        # Force all non-observable states to be within the range [1, self.N]
        # Needed because `Hs`` is 2*N column wide and `states` is N column wide.
        self.nonobstates = np.array(list(set(self.nonobstates % int(N/2))))

        # TODO: Decouple the real and imaginary observable states

        if len(self.nonobstates):
            self.isObservable = False
        else:
            self.isObservable = True       




[docs]    def checkGrossErrors(self):
        """Detect and remove gross errors

        Detect and remove gross errors through the maximum normalized
        residual test. The detection threshold is defined by
        ``r_thres``. The measurements flagged by gross errors are
        deactivated.
        """
        Ikm = np.hstack((self.Ikm.real, self.Ikm.imag))        

        Z = self.Z
        H = self.Hs
        G = self.Gf
        W_ii = np.diag(self.Ws.todense()) # Diagonal elements of R        
        M, N = H.shape

        # TODO: update this routine if the solution method changes
        r = Z - H.dot(Ikm) 
        R_ii = 1 / W_ii # Diagonal elements of R
        S_ii = np.zeros(M) # Sensitivity matrix

        for m in range(M):
            htj = H[m,:].T.todense()
            S_ii[m] = ((1 - htj.T.dot(G.solve(htj))/R_ii[m])[0,0])
 
        Omega_ii = S_ii / R_ii # computing only diagonal elements
        Omega_ii[Omega_ii == 0.] = np.inf        
        r_ = r / np.sqrt(Omega_ii)
        r_[np.isinf(r_) | np.isnan(r_)] = 0. 

        maxr_ = abs(r_).max()
        maxr_pos = np.argmax(abs(r_))

        if maxr_ > self.r_thres:
            self.hasGrossErrors = True
            #self.grossmeas += [maxr_pos]
            self.grossmeas_str += [self.Zstr[maxr_pos]]
            self.grossmeas_r_ += [maxr_]
            self.disableMeas(maxr_pos, status=None)
        else:
            self.hasGrossErrors = False                    


        


[docs]    def runDSE(self, method='branch-current', checkerrors=True, r_thres=4.0, 
               show_loadflow=True, report=False, reportmeas= True, reportv=True, 
               reporti=True, report_residuals=False):
        """Wrapper class for performing a Distributed State Estimator

        Parameters
        ----------
        method : `str`, optional
            Solution method, by default 'branch-current'
        checkerrors : `bool`, default True
            If `True`, perform gross error analysis.
        r_thres : `float`, default 4.
            Detection threshold for the Maximum Normalized Residual test
        show_loadflow : `bool`, default True
            If `True`, show the 'true' branch currents calculated from load flow.            
        report : bool, optional, default False
            Print the main summary report
        reportmeas : bool, default True
            If ``report`` is enabled, print the measurement set
        reportv : bool, default True
            If ``report`` is enabled, print the voltage profile
        reporti : bool, default True
            If ``report`` is enabled, print the current profile
        report_residuals : bool, default False
            If `True`, prints the current profile                      

        Notes
        -----
        The error detection process is activated by default.
        Note that noisy artificial measurements, as the ones generated
        by :py:func:`~pydse.dse.Dse.addMeas` can easily trigger the detection 
        threshold.
        If this is not the intended behaviour, one should disable the error
        check or increment :py:attr:`~pydse.dse.Dse.r_thres`.

        When running the DSE, the load flow 'true' solution will be automatically
        generated and added to the report, via the ``show_loadflow`` parameter.
        """
                
        self.method = method
        self.r_thres = r_thres
        self.indexBranches()

        if show_loadflow:
            self.loadflow = Loadflow(self.grid) #, loadmodel='1')
            self.loadflow.loadflow(method='zbus')            
            E_true = self.grid.E.copy()
            Ikm_true = self.getTrueCurrents()            
        else:
            self.grid.init_E()
            E_true = None
            Ikm_true = None

        self.hasGrossErrors = True # force re-check

        while self.hasGrossErrors:
            status, it, maxabserr = {
                'branch-current': self.BranchCurrentDSE,            
            }[method]()
            if checkerrors and self.isObservable:
                self.checkGrossErrors()
            else:
                self.hasGrossErrors = False

        if report:
            gridutils.summary_dse(self, status, it, maxabserr, 
                                  Ikm_true, E_true, reportmeas=reportmeas, 
                                  reportv=reportv, reporti=reporti,
                                  report_residuals=report_residuals)            




[docs]    def BranchCurrentDSE(self): 
        """ Branch-current based DSE
        
        The Branch-current based DSE (BcDSE) algorithm [1] estimates the complex 
        currents at each branch of a power grid by using a set of redundant power 
        flow or injection measurements.

        Parameters
        ----------
        report : `bool`, default False
            Print a small report after the estimation process

        Returns
        -------
        status : `bool`
            Observable status (True for observable, False otherwise)        

        References
        ----------    
        [1] https://ieeexplore.ieee.org/document/373974
        """

        self.disableGroup('|ikm|')
        self.has_nonlinear_meas = False
        # self.M = self.meter.nma() # number of active measurements
        N = self.N
        self.formH()
        self.M = self.Hs.shape[0]        
        it = 0 # branch-current iteration counter
        status = 1 # convergence status (0 to convergence)
        maxabserr = 1e20 # maximum absolute error between successive estimates        

        self.checkObservability()
        
        if not self.isObservable:
            return (status, it, maxabserr)

        iter_to_enable_imag = 2
        current_tolerance = 1e-5

        Ikm0 = np.zeros(self.nbr) # Current estimated at iteration it-1
        self.formW()               
        G = self.Hs.T.dot(self.Ws).dot(self.Hs).tocsc()
        self.Gf = sparse.linalg.splu(G) # factorized G (COLAMD - minimum degree order)

        while it < self.maxit:

            self.enableGroup('|ikm|', it == iter_to_enable_imag)
            self.formZ()
            it_wls = 0 # wls iteration counter

            while True:

                if self.has_nonlinear_meas and it >= iter_to_enable_imag and it_wls <= 3:
                    self.formH()
                    self.formW()
                    self.M = self.Hs.shape[0]
                    h = self.Hs.dot(np.hstack((self.Ikm.real,self.Ikm.imag)))
                    G = self.Hs.T.dot(self.Ws).dot(self.Hs).tocsc()
                    self.Gf = sparse.linalg.splu(G)
  
                dZ = self.Z.copy() if not self.has_nonlinear_meas else (self.Z - h)
                dIkm = self.Gf.solve(self.Hs.T.dot(self.Ws).dot(dZ))

                if not self.has_nonlinear_meas:                    
                    self.Ikm = dIkm[:N] + 1j*dIkm[N:]
                    break
                else:
                    if abs(dIkm).max() < current_tolerance or it_wls > 15:
                        break
                    else:
                        self.Ikm += (dIkm[:N] + 1j*dIkm[N:])                                                
                it_wls += 1

            maxabserr = abs(Ikm0 - self.Ikm).max()
            
            if maxabserr < self.tol and it >= iter_to_enable_imag:
                status = 0
                break                                
            self.updateVoltages()
            Ikm0 = self.Ikm.copy()
            it += 1

        return (status, it, maxabserr)
