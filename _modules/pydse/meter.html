[docs]class Meter:
    """A class to represent measurement devices

        The measurements are added using the addMeter function. Each one is 
        internally stored in the `meter` list, which is a list of Meas objects.
    """

    def __init__(self):
        self.meter = []
        self._nm = 0 # number of measurements (active or not) stored in ``meter``
        self._nm_single = 0 # number of single measurements (active or not) stored in ``meter``
        self._nmi = 0 # number of inactive measurements
        self._nmi_single = 0 # number of inactive single measurements


[docs]    def clear(self):
        self.meter.clear()
        self._nm = 0
        self._nm_single = 0
        self._nmi = 0
        self._nmi_single = 0

    

[docs]    def add(self, type, value, nodes, location, accuracy, active):
        """Add a measurement to the internal meter object

        Parameters
        ----------
        type : {'ikm','skm','i','q'}, default 'ikm'
            Type of measurement to be added
        value : `list` [`float`]
            Value of the measurement
        nodes : `str`
            Node(s) identifier
        location : `str`
            Location of measurement.
        accuracy : `float`
            Measurement accuracy (unitary)
        active : `bool`, default True
            Enable status
        """
        meas = Meas(type=type, location=location, 
                    nodes=nodes, value=value, 
                    accuracy=accuracy, active=active)
        self.meter += [meas]
        self._nm += len(nodes)
        if type in ['ikm','skm','i','q']:
            self._nm_single += 2*len(nodes)
        else:
            self._nm_single += len(nodes)

        if not active:
            self._nmi += len(nodes)
            if type in ['ikm','skm','i','q']:
                self._nmi_single += 2*len(nodes)
            else:
                self._nmi_single += len(nodes)

[docs]    def disable(self, m, nodes=None, status=False):
        """Disable a measurement of the internal meter object

        Parameters
        ----------
        m : `int`
            Index of the measurement to be deactivated in ``meter``
        nodes : `str`, default None
            Nodes to deactivate. If `None`, all nodes are deactivated
        status : {False, None}, default False
            Final disabled status. `None` means permanently disabled.
        """
        meas = self.meter[m]
        nnodes = len(meas.nodes) if not nodes else len(nodes)        
        self._nmi += meas.disable(nodes=nodes, status=status)
#         self._nmi += nnodes
        if meas.type in ['ikm', 'skm', 'i', 'q']:
            self._nmi_single += 2 * nnodes
        else:
            self._nmi_single += nnodes

    def __str__(self):
        nmeters = len(self.meter)
        if nmeters == 0:            
            return 'There are no measurement devices available'

        output = io.StringIO()
        if nmeters == 1:
            BE = 'is'
            METER = 'meter'
        else:
            BE = 'are'
            METER = 'meters'

        print('\n** Measurement set **\n') 

        style = 3 # set print style here

        if style == 1 or style == 2:

            for m in range(len(self.meter)):
                if self.meter[m].type[0].lower() == 's':
                    prunit= 'kW+jkvar'
                    prdiv = 10**3
                else:
                    prunit= 'A'
                    prdiv = 1.                
                #print(self.meter[m], file=output)
                if style == 1:
                    print("{{'type': '{}', 'location': {}, 'nodes': '{}', 'value': {}, 'accuracy': {}}}"\
                            .format(self.meter[m].type, 
                                    self.meter[m].location, 
                                    self.meter[m].nodes, 
                                    ['{:.4g}'.format(v) for v in self.meter[m].value], 
                                    self.meter[m].accuracy, file=output)
                            )
                elif style == 2:
                    print("{:>2}) type: '{}', location: {}, nodes: '{}', accuracy: {}"\
                        .format(m+1, self.meter[m].type, 
                                    self.meter[m].location, 
                                    self.meter[m].nodes, 
                                    self.meter[m].accuracy, file=output)
                        )                
                    print("    value ({}): [".format(prunit) + ', '.join(["{:.5g}{}{:.5g}"\
                        .format(v.real/prdiv,
                                '+j' if v.imag>0 else '-j',
                                abs(v.imag)/prdiv, file=output) for v in self.meter[m].value]) + ']')
        elif style == 3:
           
            # Detecting maximum bus string length
            maxloc_length = 0
            for m in range(len(self.meter)):
                sz_loc = len (' - '.join(self.meter[m].location) + ' @ ' + self.meter[m].nodes)
                if sz_loc > maxloc_length:
                    maxloc_length = sz_loc
            fmt = '{:3} {:^5s} # {:' + str(maxloc_length) + 's}   {:50s}'
            print(fmt.format('','type','location @ phase', 'value', file=output))
            print('{}'.format('-'*(50+8+maxloc_length), file=output))

            for m in range(len(self.meter)):
                if self.meter[m].type[0].lower() == 's':
                    prunit= 'kW+jkvar'
                    prdiv = 10**3
                else:
                    prunit= 'A'
                    prdiv = 1.

                value = np.array(self.meter[m].value) / prdiv
                if self.meter[m].type[0] != "|":
                    value_str = ', '.join(['{:.5g} {}{:.5g}'.format(v.real, '+j' if v.imag>0 else '-j', abs(v.imag)) for v in value])
                else:
                    value_str = ', '.join(['{:.5g}'.format(v.real) for v in value])                

                print(fmt\
                        .format(m+1, 
                                self.meter[m].type, 
                                ' - '.join(self.meter[m].location) + ' @ ' + self.meter[m].nodes, 
                                #self.meter[m].accuracy, 
                                value_str,
                                file=output
                                )
                        )

            print('{}'.format('-'*(50+8+maxloc_length), end='', file=output))

        print('\n\nThere', BE ,nmeters, METER, 'available (total of', self._nm, 'measurements):', file=output)

                               
        str_out = output.getvalue()
