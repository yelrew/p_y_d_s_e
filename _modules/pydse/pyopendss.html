    """An interface class to read OpenDSS data files

    The .dss file format is essentially a text-based command language script
    which has models for several types of power delivery, conversion and source
    elements.

    The `new` command is the basic .dss command PyDSE interprets, determining
    the type and properties of the elements to be added.
    For a full reference of the OpenDSS command language syntax, please refer to
    [1].

    Notes
    -----
    Currently, the specific implementation of the .dss codes in PyDSE is as follows:

    ================ =========================================================================
    ``Circuit``      Implements a multi-phase voltage source. PyDSE defines the substation \n 
                     voltage and angle from the basekv and angle tags. \n

    ``Line``         Models a multi-phase lines or cables, as a :math:`pi` model with series\n  
                     and shunt capacitance. The parameters can be provided as symmetrical \n
                     components or impedance matrices (most usual), by assigning it to a \n
                     ``Linecode`` object.

    ``Linecode``     Stores the impedance characteristics for lines and cables. 

    ``Transformer``  Implements a three or single phase-transformer. connected either as \n
                     as 'wye' or 'delta'.

    ``Regcontrol``   in PyDSE, a voltage regulator is considered as a transformer with \n
                     fixed ratio.

    ``Capacitor``    Multi-phase shunt (grounded) capacitor.

    ``Load``         Multi-phase 'wye' or 'delta' connected load. PyDSE implements most \n
                     of the .dss load modes, such as constant power (type 1), constant \n
                     impedance (type 2) and constant current (type 5).
    ================ =========================================================================


    See Also
    --------
    Powergrid

    References
    ----------
    ..[1] https://www.epri.com/pages/sa/opendss
    """

    def __init__(self):

        self.filepath = '' # can declare variables through self inside any method !''
        
        # Dictionaries to store power grid elements        

        self.circuit = {}       
        self.line = {}
        self.transformer = {}
        self.regcontrol = {}
        self.load = {}
        self.capacitor = {}
        self.linecode = {}                


[docs]    def read(self, filepath, redirect=False):
        """Reads an OpenDSS file (.dss) to a set dicionaries related to power elements

        The `read` method reads an OpenDSS file, parse and store its contents in
        dictionaries of `circuit`, `line`, `transformer`, `regcontrol`, `load`,
        `capacitor` and `linecode`. Currently, the function accepts the OpenDSS 
        commands `new` and `Redirect`.

        Parameters
        ----------
        filepath : `str`
            OpenDSS filepath to read
        redirect : bool, optional
            Recursive flag for the `redirect` code, by default False

        Examples
        --------
        Create a Dss object from the file 'IEEE13.dss' and print the linecodes read.

        >>> import pyopendss
        >>> dss = pyopendss.Dss(filepath='IEEE13.dss)
        >>> print(dss.linecode')               
        """        
        
        self.filepath = filepath # one can declare variables using self from any method !       

        with open(filepath) as f:
            contents = f.read()             
            contents = re.sub("/\*(.|[\r\n])*?\*/", '', contents) # removing multi-line comments            
            contents = re.sub("(!|//).*", "", contents) # removing single-line comments
            #contents = re.sub("\n~", "", contents) # handling multi-line commands
            contents = re.sub('\n\s*\n', '\n', contents) # removing blank lines
            contents = re.sub('^\n','', contents) # removing empty line
            contents = re.sub('  +',' ', contents) # removing repetitive spaces
            contents = re.sub("\n~", "", contents) # handling multi-line commands
            contents = re.sub('\s*=\s*','=',contents) # removing spaces around =
            contents = re.sub(r" ([a-zA-Z0-9%]*)=",r", \1=",contents) # adding commas before property names            
            contents = contents.split('\n') # creating string array

            # TODO: Handle multi-line comments

            #*************************

            for line in contents:

                # **************************
                # Parsing 'New' command
               
                if 'new' in line[:3].lower():

                    line = line[4:].replace('object=','').strip() # Stripping 'New ' and (possible) 'object=' string
                    sep = line.find(',')
                    parameters = line[:sep]
                    Classname, objname = parameters.split('.')
                    Classname = Classname.lower() # making Classname lowercase !!
                    data = line[sep+1:].strip()
                    
                    # TODO: Test if '=' is absent from properties. In this case, the properties 
                    # need to be defined using positional parameters!

                    if Classname != 'transformer':
                        properties = dict( properties.strip().split("=") for properties in data.split(",") )
                    else:
                        # Separating identical transformer tags
                        properties = {}
                        tags = [ properties.strip().split("=") for properties in data.split(",") ]
                        for tag, value in tags:
                            c = 1 # counter
                            tag_ = tag
                            while True:                                
                                if tag not in properties:
                                    properties[tag] = value
                                    break
                                tag = tag_ + '.' + str(c)                   

                    properties = dict((k.lower(), v) for k,v in properties.items()) # converting keys to lowercase

                    #TODO: Currently, the keywork "Object=" before the Class name is not handled ! 
                    
                    if Classname == 'circuit': self.circuit[objname] = properties # Load
                    elif Classname == 'line': self.line[objname] = properties
                    elif Classname == 'linecode': self.linecode[objname] = properties # Linecode
                    elif Classname == 'transformer': self.transformer[objname] = properties                        
                    elif Classname == 'regcontrol': self.regcontrol[objname] = properties                        
                    elif Classname == 'load': self.load[objname] = properties # Load
                    elif Classname == 'capacitor': self.capacitor[objname] = properties # Load

                if 'redirect' in line[:8].lower():
                    from os.path import dirname
                    extfile = line.strip().split()[1]
                    self.read(dirname(filepath) + '/' + extfile, redirect=True)
                    self.filepath = filepath

            # Post-processing data
            if not redirect:
                self.post_process_circuit()
                self.post_process_line()
                self.post_process_linecode() # convert string matrices to numpy arrays
                self.post_process_capacitor()
                self.post_process_transformer()


    

[docs]    def post_process_circuit(self):
        """Additional parsing of ``circuit`` codes
        """
        for id_ in self.circuit:
            if 'bus1' not in self.circuit[id_]:
                self.circuit[id_]['bus1'] = 'sourcebus'
            if 'phases' not in self.circuit[id_]:
                self.circuit[id_]['phases'] = '3'
            nph = int (self.circuit[id_]['phases'])  
            if '.' not in self.circuit[id_]['bus1']:
                self.circuit[id_]['bus1'] += '.1.2.3'[:nph*2]




[docs]    def post_process_line(self):
        """Additional parsing of ``line`` codes
        """
        for id_ in self.line:
            # filling default length
            if 'length' not in self.line[id_]:
                self.line[id_]['length'] = '1.'

            if 'units' not in self.line[id_]:
                self.line[id_]['units'] = None

            nph = int (self.line[id_]['phases'])  

            # filling default phase sequence
            if '.' not in self.line[id_]['bus1']:
                self.line[id_]['bus1'] += '.1.2.3'[:nph*2]

            # filling default phase sequence
            if '.' not in self.line[id_]['bus2']:
                self.line[id_]['bus2'] += '.1.2.3'[:nph*2]

[docs]    def post_process_linecode(self): # transform string matrices into numpy arrays (adds new properties to linecode[][])
        """Additional parsing of ``linecode`` codes
        """
        
        code = list(self.linecode) # list with linecodes

        for i in range (len(code)):

            subkey = [s for s in list(self.linecode[code[i]].keys()) if "matrix" in s] # list of 'matrix'-type subkeys
            for j in range(len(subkey)):
                rowmatrix = re.sub('[\(\)\[\]]', '', self.linecode[code[i]][subkey[j]]) # removing parentheses
                rowmatrix = rowmatrix.split('|') # creating string array with matrix rows               
                nr = len(rowmatrix) # number of rows in the 'matrix' property
                linecode_np = np.zeros((nr,nr))

                nel = 0 # total number of elements read
                for r in range(nr):
                    row = np.fromstring(rowmatrix[r], sep=' ')
                    linecode_np[r,:len(row)] = row
                    nel += len(row)

                # complete symmetric matrix
                nphases = int(self.linecode[code[i]]['nphases'])
                if nel < nphases*nphases:
                    linecode_np += np.tril(linecode_np,-1).T
                self.linecode[code[i]][subkey[j].lower() +'_np'] = linecode_np




[docs]    def post_process_transformer(self):
        """Additional parsing of ``transformer`` codes
        """

        # Filling keys 'bus1' and 'bus2' for transformers
        for id_ in self.transformer:
            # Handling arrays of bus tags
            transformer = self.transformer[id_]
            if 'buses' in transformer:
                if len(transformer['buses'].split()) == 3:
                    print('Can''t handle three-wind transformers')
                    exit(1)
                bus1, bus2 = re.sub('[\(\)\[\]]', '', transformer['buses']).split()
                self.transformer[id_]['bus1'] = bus1
                self.transformer[id_]['bus2'] = bus2
            else:
                self.transformer[id_]['bus1'] = transformer['bus']
                self.transformer[id_]['bus2'] = transformer['bus.1']
            
            # Filling default phase sequence
            nph  = int(transformer['phases'])
            if '.' not in transformer['bus1']:
                self.transformer[id_]['bus1'] += '.1.2.3'[:nph*2]
            if '.' not in self.transformer[id_]['bus2']:
                self.transformer[id_]['bus2'] += '.1.2.3'[:nph*2]   
                     
            # Removing []()"' from arrays
            for key, value in self.transformer[id_].items():
                self.transformer[id_][key] = re.sub('[\(\)\[\]"\']', '', value)

            # Filling default options
            if 'conns' not in transformer and 'conn' not in transformer:
                self.transformer[id_]['conns'] = 'wye wye'
            if 'taps' not in transformer and 'tap' not in transformer:
                self.transformer[id_]['taps'] = '1.0 1.0'

    def yfromcsv(filepath):
        """Retrieves an Admittance matrix generated by OpenDSS from a csv file
        """

        with open(filepath) as f:

            lines = f.readlines()
            dim = int(lines[0].split(',')[0])
            Y = np.zeros((dim,dim),dtype=complex)

            for l in range(1,len(lines)):   # first line has the array dimension
                line = re.sub('(.*?)\s*,\s*\+j\s*(.*?)\s*,',r'\1+\2j ,',lines[l]) # joining real and imaginary parts
                line = re.sub('\+-','-',line)
                virg = line.find(',')
                bus = line[:virg].replace('"','')
                line = line[virg+1:].replace(' ','')
                line_fields = line.split(',')
                
                for f in range (dim):
                    Y[l-1,f] = complex(line_fields[f])

            return Y

    def read_vln_vll(filepath, source):
        """Read voltage output generated by OpenDSS

        Parameters
        ----------
        filepath : `str`
            OpenDSS file
        source : `str`
            Name of the source bus

        Returns
        -------
        vv : `pandas.Dataframe`
            Dataframe holding line-to-neutral and line-to-line voltages.
        """
        from io import StringIO
        dots = '.'*(len(source)-4)        
        sio = StringIO(open(filepath).read().replace(dots,''))        

        vv = pd.read_csv(sio, sep='\s+', skiprows=[*range(4)], header = None,
                         names = ['bus', 'node', 'vln-kv', '1', 'vln-ang', 'vln-pu', 'basekv',
                                  'node-node', 'vll-kv', '2', 'vll-ang', 'vll-pu'],
                         usecols=[*range(3), *range(4,9), 10, 11 ])
        
        return vv # database with phase and line voltages
